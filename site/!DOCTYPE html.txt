    <script>
        // ===== SEULEMENT LES MAPS QUI FONCTIONNENT =====
        const maps = [
            { name: 'Paris', city: 'France', flag: '🇫🇷', url: 'https://ss.randomkzn.com/paris' },
            { name: 'Londres', city: 'UK', flag: '🇬🇧', url: 'https://ss.randomkzn.com/london' },
            { name: 'Le Caire', city: 'Égypte', flag: '🇪🇬', url: 'https://ss.randomkzn.com/cairo' },
            { name: 'Pékin', city: 'Chine', flag: '🇨🇳', url: 'https://ss.randomkzn.com/beijing' },
            { name: 'Mexico', city: 'Mexique', flag: '🇲🇽', url: 'https://ss.randomkzn.com/mexico' }
        ];

        // ===== SCORES =====
        let totalScore = parseInt(localStorage.getItem('nayloxTotal')) || 0;
        let highScore = parseInt(localStorage.getItem('nayloxHigh')) || 0;

        document.getElementById('totalScore').textContent = totalScore;
        document.getElementById('highScore').textContent = highScore;

        // ===== AFFICHER LES MAPS =====
        const grid = document.getElementById('mapsGrid');

        maps.forEach(map => {
            const card = document.createElement('a');
            card.className = 'map-card';
            card.href = map.url;
            card.target = '_blank';

            card.innerHTML = `
                <span class="flag">${map.flag}</span>
                <div class="name">${map.name}</div>
                <div class="city">${map.city}</div>
                <span class="badge">▶ JOUER</span>
                <span class="status">✅</span>
            `;

            card.addEventListener('click', () => {
                totalScore += 1;
                localStorage.setItem('nayloxTotal', totalScore);
                document.getElementById('totalScore').textContent = totalScore;
            });

            grid.appendChild(card);
        });
    </script>
